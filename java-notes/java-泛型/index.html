public class Em<T> {
    private T num;
    private T name;

    public T getNum() {
        return num;
    }

    public void setNum(T num) {
        this.num = num;
    }

    public T getName() {
        return name;
    }

    public void setName(T name) {
        this.name = name;
    }

    public Em() {

    }

    public Em(T num, T name) {
        this.num = num;
        this.name = name;
    }

    @Override
    public String toString() {
        return "Em{" +
                "num=" + num +
                ", name=" + name +
                '}';
    }
}

import java.util.Comparator;

public class EmComparable implements Comparator<Em<Integer>> {

    @Override
    public int compare(Em<Integer> o1,Em<Integer>o2) {
        if(o1.getNum()>o2.getNum())
            return 1;
        else if(o1.getNum()<o2.getNum())
            return -1;
        else
            return 0;
    }
}
